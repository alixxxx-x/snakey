import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.Label;
import javafx.scene.control.TextField;
import javafx.scene.layout.TilePane;
import javafx.scene.layout.VBox;
import javafx.scene.paint.Color;
import javafx.scene.layout.BorderPane;
import javafx.scene.layout.HBox;
import javafx.scene.layout.Pane;
import javafx.scene.shape.Circle;
import javafx.scene.paint.Color;
import javafx.scene.shape.Rectangle;
import javafx.scene.layout.Priority;
import javafx.stage.Stage;
import javafx.scene.layout.Region;
import javafx.scene.layout.StackPane;
import javafx.geometry.Insets;
import javafx.geometry.Pos;

public class HelloFX extends Application {

    @Override
    public void start(Stage stage) {

        
        

        Circle circle = new Circle(20);
        circle.setFill(Color.RED);
        circle.setCenterX(200);
        circle.setCenterY(200);

        Pane rec = new Pane(circle);
        rec.setPrefSize(400, 400);
        rec.setStyle("-fx-border-color: black;");

        StackPane container = new StackPane(rec);
        container.setAlignment(Pos.CENTER);
        container.setMaxSize(400,400);
        

        Button up = new Button("up");
        Button down = new Button("down");
        Button left = new Button("left");
        Button right = new Button("right");

        HBox buttons = new HBox();
        buttons.getChildren().addAll(up,down,left,right);
        buttons.setAlignment(Pos.CENTER);

        VBox root = new VBox();
        root.getChildren().addAll(container,buttons);
        root.setAlignment(Pos.CENTER);
        Scene scene = new Scene(root,700,700);



        up.setOnAction(e -> 
            circle.setCenterY(circle.getCenterY() - 10)
        );
        down.setOnAction(e -> 
            circle.setCenterY(circle.getCenterY() + 10)
        );
        left.setOnAction(e -> 
            circle.setCenterX(circle.getCenterX() - 10)
        );
        right.setOnAction(e -> 
            circle.setCenterX(circle.getCenterX() + 10)
        );
        

        stage.setScene(scene);
        stage.setTitle("aymen");
        stage.show();



    }

    public static void main(String[] args) {
        launch();
    }
}
